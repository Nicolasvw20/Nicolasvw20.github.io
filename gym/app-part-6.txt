  async function promptInstall() {
    if(deferredInstall){ deferredInstall.prompt(); await deferredInstall.userChoice; deferredInstall=null; }
    else alert('iPhone: open in Safari → Share → Add to Home Screen. Android: open in Chrome → menu → Install app.');
  }

  document.querySelectorAll('[data-view]').forEach(btn=>btn.addEventListener('click',()=>setView(btn.dataset.view)));

    document.querySelectorAll('.block').forEach(()=>{});
    if(state.timer.running) startTimerInterval();
    if('serviceWorker' in navigator) navigator.serviceWorker.register('service-worker.js').catch(()=>{});
  });
